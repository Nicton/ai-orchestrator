## Запуск проекта

### Требования
- Node.js >= 24.4.0 (см. `.nvmrc`)
- PostgreSQL
- Redis

### Команды

```bash
# API сервер (порт 3013)
npm run dev

# API сервер с авто-перезагрузкой
npm run dev:watch

# API сервер с отладкой (порт 9229)
npm run dev:watch-debug

# Worker (порт 3018)
npm run worker

# Worker с авто-перезагрузкой
npm run worker:watch

# Мониторинг очередей
npm run kue-dashboard    # http://localhost:9128
```

### Тесты и линтер

```bash
npm test                  # Jest
npm run test:watch        # Jest в watch-режиме
npm run test:integration  # Интеграционные тесты (отдельная БД)
npm run eslint:backend    # Линтинг + автоисправление
npm run eslint:edited     # Только изменённые файлы (pre-commit hook)
```

## Аутентификация и авторизация

### JWT-токен

```
Запрос с заголовком: Authorization: Bearer <token>
    ↓
express-jwt: верифицирует подпись, проверяет срок
    ↓
Загрузка User из БД
    ↓
req.user         → User instance
req.user.account → Account instance
req.user.roles_list → ['api', 'self_admin', ...]
```

Отзыв токенов хранится в Redis. При logout — токен добавляется в blacklist.

### ACL-система

```javascript
// app/lib/acl.js

// Защита маршрута ролью
router.get('/shipments', authByRolesMiddleware(['api', 'admin']), controller);

// Проверка права внутри сервиса
if (!user.roles_list.includes('admin')) {
    throw new ForbiddenError('Access denied');
}
```

Доступ фильтруется на уровне SQL-запросов:
- Shipper видит только свои данные (`shipper_id = ?`)
- Carrier видит только свои слоты и связанные перевозки
- Разрешения задаются через `ShipperACL` / `CarrierACL` по divisions

## Паттерны кода

### Структура сервиса

```javascript
// app/services/example.js
const db = require('../models')();
const { buildLogger } = require('../lib/log');
const { BadRequestError, NotFoundError } = require('../errors');
const resolvers = require('./resolvers');

const logger = buildLogger('example-service');

const findEntity = (id, user) => {
    return resolvers.loadEntity(id)
        .then(entity => {
            if (!entity) throw new NotFoundError('Entity not found');
            return entity;
        });
};

module.exports = { findEntity, createEntity, updateEntity };
```

### Кастомные ошибки

```javascript
const { BadRequestError, ForbiddenError, NotFoundError } = require('../errors');

throw new BadRequestError('Invalid input');    // → HTTP 400
throw new ForbiddenError('Access denied');     // → HTTP 403
throw new NotFoundError('Not found');          // → HTTP 404
```

### Работа с БД (Sequelize v3)

```javascript
// Поиск с включением связанных моделей
db.Shipment.find({
    where: { id: shipmentId, shipper_id: user.account.shipper_id },
    include: [
        { model: db.Carrier, as: 'carrier' },
        { model: db.ShipperDivision, as: 'division' }
    ]
})

// Транзакция
db.sequelize.transaction(t => {
    return db.Shipment.create({ ...data }, { transaction: t })
        .then(shipment => db.ShipmentRequest.update(
            { status: 'confirmed' },
            { where: { id: shipment.request_id }, transaction: t }
        ));
});

// Мягкое удаление (paranoid: true)
// Sequelize добавляет WHERE deleted_at IS NULL автоматически
shipment.destroy(); // Устанавливает deleted_at, не удаляет физически
```

### Логирование

```javascript
const { buildLogger } = require('../lib/log');
const logger = buildLogger('shipments-service');

logger.info('Creating shipment', { userId, carrierId });
```

### Promise-цепочки (legacy стиль)

Codebase использует Promise chains (не async/await в старых файлах):

```javascript
return loadShipment(id)
    .then(shipment => validateShipment(shipment))
    .then(shipment => bookWithCarrier(shipment))
    .then(result => updateStatus(result))
    .catch(err => {
        logger.error('Booking error', { err, shipmentId: id });
        throw err;
    });
```

### Валидация (Joi)

```javascript
const Joi = require('joi');

const schema = Joi.object({
    carrierId: Joi.number().integer().required(),
    pickupDate: Joi.date().iso().required(),
    weight: Joi.number().positive()
});

const { error, value } = schema.validate(req.body);
if (error) throw new BadRequestError(error.message);
```

## Структура контроллера

```javascript
// app/controllers/api/shipments.js
const shipmentsService = require('../../services/shipments');

module.exports = {
    list: (req, res, next) => {
        const { user, query } = req;
        shipmentsService.listShipments(user, query)
            .then(result => res.json(result))
            .catch(next);
    },

    create: (req, res, next) => {
        shipmentsService.createShipment(req.user, req.body)
            .then(shipment => res.status(201).json(shipment))
            .catch(next);
    }
};
```

## Структура маршрутов

```javascript
// app/routes/shipments.js
const express = require('express');
const router = express.Router();
const controller = require('../controllers/api/shipments');
const { authByRolesMiddleware } = require('../lib/middleware/auth');

router.get('/', authByRolesMiddleware(['api']), controller.list);
router.post('/', authByRolesMiddleware(['api']), controller.create);
router.get('/:id', authByRolesMiddleware(['api']), controller.get);

module.exports = router;
```

## Добавление новой фичи — чеклист

1. **Модель** (если нужна новая таблица): `app/models/new_entity.js`
2. **Сервис**: `app/services/new_feature.js` — бизнес-логика
3. **Контроллер**: `app/controllers/api/new_feature.js` — HTTP-обработка
4. **Маршрут**: `app/routes/new_feature.js` — регистрация эндпоинта
5. **Подключить маршрут** в `app/lib/index.js`
6. **Тест**: `test/new_feature.spec.js`
7. **Если async** → добавить handler в воркер (`05_ВОРКЕР_И_ОЧЕРЕДИ.md`)
8. **Если интеграция** → следовать паттерну (`04_ИНТЕГРАЦИИ.md`)
